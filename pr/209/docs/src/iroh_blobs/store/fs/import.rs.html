1//! The entire logic for importing data from three different sources: bytes, byte stream, and file path.
2//!
3//! For bytes, the size is known in advance. But they might still have to be persisted to disk if they are too large.
4//! E.g. you add a 1GB bytes to the store, you still want this to end up on disk.
5//!
6//! For byte streams, the size is not known in advance.
7//!
8//! For file paths, the size is known in advance. This is also the only case where we might reference the data instead
9//! of copying it.
10//!
11//! The various ..._task fns return an `Option<ImportEntry>`. If import fails for whatever reason, the error goes
12//! to the requester, and the task returns None.

54/// An import source.
55///
56/// It must provide a way to read the data synchronously, as well as the size
57/// and the file location.
58///
59/// This serves as an intermediate result between copying and outboard computation.
60pub enum ImportSource {
61    TempFile(PathBuf, File, u64),
62    External(PathBuf, File, u64),
63    Memory(Bytes),
64}
65
66impl std::fmt::Debug for ImportSource {
67    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
68        match self {
69            Self::TempFile(path, _, size) => {
70                f.debug_tuple("TempFile").field(path).field(size).finish()
71            }
72            Self::External(path, _, size) => {
73                f.debug_tuple("External").field(path).field(size).finish()
74            }
75            Self::Memory(data) => f.debug_tuple("Memory").field(&data.len()).finish(),
76        }
77    }
78}

80impl ImportSource {
81    pub fn fmt_short(&self) -> String {
82        match self {
83            Self::TempFile(path, _, _) => format!("TempFile({})", path.display()),
84            Self::External(path, _, _) => format!("External({})", path.display()),
85            Self::Memory(data) => format!("Memory({})", data.len()),
86        }
87    }
88
89    fn is_mem(&self) -> bool {
90        matches!(self, Self::Memory(_))
91    }
92
93    /// A reader for the import source.
94    fn read(&self) -> MemOrFile<std::io::Cursor<&[u8]>, &File> {
95        match self {
96            Self::TempFile(_, file, _) => MemOrFile::File(file),
97            Self::External(_, file, _) => MemOrFile::File(file),
98            Self::Memory(data) => MemOrFile::Mem(std::io::Cursor::new(data.as_ref())),
99        }
100    }
101
102    /// The size of the import source.
103    fn size(&self) -> u64 {
104        match self {
105            Self::TempFile(_, _, size) => *size,
106            Self::External(_, _, size) => *size,
107            Self::Memory(data) => data.len() as u64,
108        }
109    }
110}

112/// An import entry.
113///
114/// This is the final result of an import operation. It gets passed to the store
115/// for integration.
116///
117/// The store can assume that the outboard, if on disk, is in a location where
118/// it can be moved to the final location (basically it needs to be on the same device).
119pub struct ImportEntry {
120    pub hash: Hash,
121    pub format: BlobFormat,
122    pub scope: Scope,
123    pub source: ImportSource,
124    pub outboard: MemOrFile<Bytes, PathBuf>,
125}
126
127impl std::fmt::Debug for ImportEntry {
128    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
129        f.debug_struct("ImportEntry")
130            .field("hash", &self.hash)
131            .field("format", &self.format)
132            .field("scope", &self.scope)
133            .field("source", &DD(self.source.fmt_short()))
134            .field("outboard", &DD(self.outboard.fmt_short()))
135            .finish()
136    }
137}

159/// Start a task to import from a [`Bytes`] in memory.
160#[instrument(skip_all, fields(data = cmd.data.len()))]
161pub async fn import_bytes(cmd: ImportBytesMsg, ctx: Arc<TaskContext>) {
162    let size = cmd.data.len() as u64;
163    if ctx.options.is_inlined_all(size) {
164        import_bytes_tiny_outer(cmd, ctx).await;
165    } else {
166        let request = ImportByteStreamRequest {
167            format: cmd.format,
168            scope: cmd.scope,
169        };
170        let stream = stream::iter(Some(Ok(cmd.data.clone())));
171        import_byte_stream_mid(request, cmd.tx, cmd.span, stream, ctx).await;
172    }
173}

175async fn import_bytes_tiny_outer(mut cmd: ImportBytesMsg, ctx: Arc<TaskContext>) {
176    match import_bytes_tiny_impl(cmd.inner, &mut cmd.tx).await {
177        Ok(entry) => {
178            let entry = ImportEntryMsg {
179                inner: entry,
180                tx: cmd.tx,
181                rx: cmd.rx,
182                span: cmd.span,
183            };
184            ctx.internal_cmd_tx.send(entry.into()).await.ok();
185        }
186        Err(cause) => {
187            cmd.tx.send(cause.into()).await.ok();
188        }
189    }
190}

192async fn import_bytes_tiny_impl(
193    cmd: ImportBytesRequest,
194    tx: &mut mpsc::Sender<AddProgressItem>,
195) -> io::Result<ImportEntry> {
196    let size = cmd.data.len() as u64;
197    // send the required progress events
198    // AddProgressItem::Done will be sent when finishing the import!
199    tx.send(AddProgressItem::Size(size)).await?;
200    tx.send(AddProgressItem::CopyDone).await?;
201    Ok(if raw_outboard_size(size) == 0 {
202        // the thing is so small that it does not even need an outboard
203        ImportEntry {
204            hash: Hash::new(&cmd.data),
205            format: cmd.format,
206            scope: cmd.scope,
207            source: ImportSource::Memory(cmd.data),
208            outboard: MemOrFile::empty(),
209        }
210    } else {
211        // we still know that computing the outboard will be super fast
212        let outboard = PreOrderMemOutboard::create(&cmd.data, IROH_BLOCK_SIZE);
213        ImportEntry {
214            hash: outboard.root.into(),
215            format: cmd.format,
216            scope: cmd.scope,
217            source: ImportSource::Memory(cmd.data),
218            outboard: MemOrFile::Mem(Bytes::from(outboard.data)),
219        }
220    })
221}

223#[instrument(skip_all)]
224pub async fn import_byte_stream(cmd: ImportByteStreamMsg, ctx: Arc<TaskContext>) {
225    let stream = into_stream(cmd.rx);
226    import_byte_stream_mid(cmd.inner, cmd.tx, cmd.span, stream, ctx).await
227}

229fn into_stream(
230    mut rx: mpsc::Receiver<ImportByteStreamUpdate>,
231) -> impl Stream<Item = io::Result<Bytes>> {
232    Gen::new(|co| async move {
233        loop {
234            match rx.recv().await {
235                Ok(Some(ImportByteStreamUpdate::Bytes(data))) => {
236                    co.yield_(Ok(data)).await;
237                }
238                Ok(Some(ImportByteStreamUpdate::Done)) => {
239                    break;
240                }
241                Ok(None) => {
242                    co.yield_(Err(io::ErrorKind::UnexpectedEof.into())).await;
243                    break;
244                }
245                Err(e) => {
246                    co.yield_(Err(e.into())).await;
247                    break;
248                }
249            }
250        }
251    })
252}

277async fn import_byte_stream_impl(
278    cmd: ImportByteStreamRequest,
279    tx: &mut mpsc::Sender<AddProgressItem>,
280    stream: impl Stream<Item = io::Result<Bytes>> + Unpin,
281    options: Arc<Options>,
282) -> io::Result<ImportEntry> {
283    let ImportByteStreamRequest { format, scope } = cmd;
284    let import_source = get_import_source(stream, tx, &options).await?;
285    tx.send(AddProgressItem::Size(import_source.size())).await?;
286    tx.send(AddProgressItem::CopyDone).await?;
287    compute_outboard(import_source, format, scope, options, tx).await
288}

290async fn get_import_source(
291    stream: impl Stream<Item = io::Result<Bytes>> + Unpin,
292    tx: &mut mpsc::Sender<AddProgressItem>,
293    options: &Options,
294) -> io::Result<ImportSource> {
295    let mut stream = stream.fuse();
296    let mut peek = SmallVec::<[_; 2]>::new();
297    let Some(first) = stream.next().await.transpose()? else {
298        return Ok(ImportSource::Memory(Bytes::new()));
299    };
300    match stream.next().await.transpose()? {
301        Some(second) => {
302            peek.push(Ok(first));
303            peek.push(Ok(second));
304        }
305        None => {
306            let size = first.len() as u64;
307            if options.is_inlined_data(size) {
308                return Ok(ImportSource::Memory(first));
309            }
310            peek.push(Ok(first));
311        }
312    };
313    // todo: if both first and second are giant, we might want to write them to disk immediately
314    let mut stream = stream::iter(peek).chain(stream);
315    let mut size = 0;
316    let mut data = Vec::new();
317    let mut disk = None;
318    while let Some(chunk) = stream.next().await {
319        let chunk = chunk?;
320        size += chunk.len() as u64;
321        if size > options.inline.max_data_inlined {
322            let temp_path = options.path.temp_file_name();
323            trace!("writing to temp file: {:?}", temp_path);
324            let mut file = fs::OpenOptions::new()
325                .read(true)
326                .write(true)
327                .create(true)
328                .truncate(true)
329                .open(&temp_path)?;
330            file.write_all(&data)?;
331            file.write_all(&chunk)?;
332            data.clear();
333            disk = Some((file, temp_path));
334            break;
335        } else {
336            data.extend_from_slice(&chunk);
337        }
338        // todo: don't send progress for every chunk if the chunks are small?
339        tx.try_send(AddProgressItem::CopyProgress(size)).await?;
340    }
341    Ok(if let Some((mut file, temp_path)) = disk {
342        while let Some(chunk) = stream.next().await {
343            let chunk = chunk?;
344            file.write_all(&chunk)?;
345            size += chunk.len() as u64;
346            tx.send(AddProgressItem::CopyProgress(size)).await?;
347        }
348        ImportSource::TempFile(temp_path, file, size)
349    } else {
350        ImportSource::Memory(data.into())
351    })
352}

354#[derive(ref_cast::RefCast)]
355#[repr(transparent)]
356struct OutboardProgress(mpsc::Sender<AddProgressItem>);
357
358impl Sink<ChunkNum> for OutboardProgress {
359    type Error = irpc::channel::SendError;
360
361    async fn send(&mut self, offset: ChunkNum) -> std::result::Result<(), Self::Error> {
362        // if offset.0 % 1024 != 0 {
363        //     return Ok(());
364        // }
365        self.0
366            .try_send(AddProgressItem::OutboardProgress(offset.to_bytes()))
367            .await?;
368        Ok(())
369    }
370}

372async fn compute_outboard(
373    source: ImportSource,
374    format: BlobFormat,
375    scope: Scope,
376    options: Arc<Options>,
377    tx: &mut mpsc::Sender<AddProgressItem>,
378) -> io::Result<ImportEntry> {
379    let size = source.size();
380    let tree = BaoTree::new(size, IROH_BLOCK_SIZE);
381    let root = bao_tree::blake3::Hash::from_bytes([0; 32]);
382    let outboard_size = raw_outboard_size(size);
383    let send_progress = OutboardProgress::ref_cast_mut(tx);
384    let mut data = source.read();
385    data.rewind()?;
386    let (hash, outboard) = if outboard_size > options.inline.max_outboard_inlined {
387        // outboard will eventually be stored as a file, so compute it directly to a file
388        // we don't know the hash yet, so we need to create a temp file
389        let outboard_path = options.path.temp_file_name();
390        trace!("Creating outboard file in {}", outboard_path.display());
391        // we don't need to read from this file!
392        let mut outboard_file = File::create(&outboard_path)?;
393        let mut outboard = PreOrderOutboard {
394            tree,
395            root,
396            data: &mut outboard_file,
397        };
398        init_outboard(data, &mut outboard, send_progress).await??;
399        (outboard.root, MemOrFile::File(outboard_path))
400    } else {
401        // outboard will be stored in memory, so compute it to a memory buffer
402        trace!("Creating outboard in memory");
403        let mut outboard_file: Vec<u8> = Vec::new();
404        let mut outboard = PreOrderOutboard {
405            tree,
406            root,
407            data: &mut outboard_file,
408        };
409        init_outboard(data, &mut outboard, send_progress).await??;
410        (outboard.root, MemOrFile::Mem(Bytes::from(outboard_file)))
411    };
412    Ok(ImportEntry {
413        hash: hash.into(),
414        format,
415        scope,
416        source,
417        outboard,
418    })
419}

439async fn import_path_impl(
440    cmd: ImportPathRequest,
441    tx: &mut mpsc::Sender<AddProgressItem>,
442    options: Arc<Options>,
443) -> io::Result<ImportEntry> {
444    let ImportPathRequest {
445        path,
446        mode,
447        format,
448        scope: batch,
449    } = cmd;
450    if !path.is_absolute() {
451        return Err(io::Error::new(
452            io::ErrorKind::InvalidInput,
453            "path must be absolute",
454        ));
455    }
456    if !path.is_file() && !path.is_symlink() {
457        return Err(io::Error::new(
458            io::ErrorKind::InvalidInput,
459            "path is not a file or symlink",
460        ));
461    }
462
463    let size = path.metadata()?.len();
464    tx.send(AddProgressItem::Size(size)).await?;
465    let import_source = if size <= options.inline.max_data_inlined {
466        let data = std::fs::read(path)?;
467        tx.send(AddProgressItem::CopyDone).await?;
468        ImportSource::Memory(data.into())
469    } else if mode == ImportMode::TryReference {
470        // reference where it is. We are going to need the file handle to
471        // compute the outboard, so open it here. If this fails, the import
472        // can't proceed.
473        let file = OpenOptions::new().read(true).open(&path)?;
474        ImportSource::External(path, file, size)
475    } else {
476        let temp_path = options.path.temp_file_name();
477        // todo: if reflink works, we don't need progress.
478        // But if it does not, it might take a while and we won't get progress.
479        let res = reflink_or_copy_with_progress(&path, &temp_path, size, tx).await?;
480        trace!(
481            "imported {} to {}, {res:?}",
482            path.display(),
483            temp_path.display()
484        );
485        // copy from path to temp_path
486        let file = OpenOptions::new().read(true).open(&temp_path)?;
487        tx.send(AddProgressItem::CopyDone).await?;
488        ImportSource::TempFile(temp_path, file, size)
489    };
490    compute_outboard(import_source, format, batch, options, tx).await
491}

507    async fn drain<T: RpcMessage>(mut recv: mpsc::Receiver<T>) -> TestResult<Vec<T>> {
508        let mut res = Vec::new();
509        while let Some(item) = recv.recv().await? {

524    fn chunk_bytes(data: Bytes, chunk_size: usize) -> impl Iterator<Item = Bytes> {
525        assert!(chunk_size > 0, "Chunk size must be positive");
526        (0..data.len())
527            .step_by(chunk_size)
528            .map(move |i| data.slice(i..std::cmp::min(i + chunk_size, data.len())))
529    }
