43/// Storage for complete blobs. There is no longer any uncertainty about the
44/// size, so we don't need a sizes file.
45///
46/// Writing is not possible but also not needed, since the file is complete.
47/// This covers all combinations of data and outboard being in memory or on
48/// disk.
49///
50/// For the memory variant, it does reading in a zero copy way, since storage
51/// is already a `Bytes`.
52#[derive(Default)]
53pub struct CompleteStorage {
54    /// data part, which can be in memory or on disk.
55    pub data: MemOrFile<Bytes, FixedSize<File>>,
56    /// outboard part, which can be in memory or on disk.
57    pub outboard: MemOrFile<Bytes, File>,
58}

94/// Read from the given file at the given offset, until end of file or max bytes.
95fn read_to_end(file: impl ReadAt, offset: u64, max: usize) -> io::Result<Bytes> {
96    let mut res = BytesMut::new();
97    let mut buf = [0u8; 4096];
98    let mut remaining = max;
99    let mut offset = offset;
100    while remaining > 0 {
101        let end = buf.len().min(remaining);
102        let read = file.read_at(offset, &mut buf[..end])?;
103        if read == 0 {
104            // eof
105            break;
106        }
107        res.extend_from_slice(&buf[..read]);
108        offset += read as u64;
109        remaining -= read;
110    }
111    Ok(res.freeze())
112}

114fn max_offset(batch: &[BaoContentItem]) -> u64 {
115    batch
116        .iter()
117        .filter_map(|item| match item {
118            BaoContentItem::Leaf(leaf) => {
119                let len = leaf.data.len().try_into().unwrap();
120                let end = leaf
121                    .offset
122                    .checked_add(len)
123                    .expect("u64 overflow for leaf end");
124                Some(end)
125            }
126            _ => None,
127        })
128        .max()
129        .unwrap_or(0)
130}

132/// A file storage for an incomplete bao file.
133#[derive(Debug)]

136    outboard: std::fs::File,
137    sizes: std::fs::File,
138    bitfield: Bitfield,
139}

141impl PartialFileStorage {
142    pub fn bitfield(&self) -> &Bitfield {
143        &self.bitfield
144    }
145
146    pub(super) fn sync_all(&self, bitfield_path: &Path) -> io::Result<()> {
147        self.data.sync_all()?;
148        self.outboard.sync_all()?;
149        self.sizes.sync_all()?;
150        // only write the bitfield if the syncs were successful
151        trace!(
152            "writing bitfield {:?} to {}",
153            self.bitfield,
154            bitfield_path.display()
155        );
156        write_checksummed(bitfield_path, &self.bitfield)?;
157        Ok(())
158    }
159
160    fn load(hash: &Hash, options: &PathOptions) -> io::Result<Self> {
161        let bitfield_path = options.bitfield_path(hash);
162        let data = create_read_write(options.data_path(hash))?;
163        let outboard = create_read_write(options.outboard_path(hash))?;
164        let sizes = create_read_write(options.sizes_path(hash))?;
165        let bitfield = match read_checksummed_and_truncate(&bitfield_path) {
166            Ok(bitfield) => bitfield,
167            Err(cause) => {
168                trace!(
169                    "failed to read bitfield for {} at {}: {:?}",
170                    hash.to_hex(),
171                    bitfield_path.display(),
172                    cause
173                );
174                trace!("reconstructing bitfield from outboard");
175                let size = read_size(&sizes).ok().unwrap_or_default();
176                let outboard = PreOrderOutboard {
177                    data: &outboard,
178                    tree: BaoTree::new(size, IROH_BLOCK_SIZE),
179                    root: blake3::Hash::from(*hash),
180                };
181                let mut ranges = ChunkRanges::empty();
182                for range in bao_tree::io::sync::valid_ranges(outboard, &data, &ChunkRanges::all())
183                    .into_iter()
184                    .flatten()
185                {
186                    ranges |= ChunkRanges::from(range);
187                }
188                info!("reconstructed range is {:?}", ranges);
189                Bitfield::new(ranges, size)
190            }
191        };
192        Ok(Self {
193            data,
194            outboard,
195            sizes,
196            bitfield,
197        })
198    }
199
200    fn into_complete(
201        self,
202        size: u64,
203        options: &Options,
204    ) -> io::Result<(CompleteStorage, EntryState<Bytes>)> {
205        let outboard_size = raw_outboard_size(size);
206        let (data, data_location) = if options.is_inlined_data(size) {
207            let data = read_to_end(&self.data, 0, size as usize)?;
208            (MemOrFile::Mem(data.clone()), DataLocation::Inline(data))
209        } else {
210            (
211                MemOrFile::File(FixedSize::new(self.data, size)),
212                DataLocation::Owned(size),
213            )
214        };
215        let (outboard, outboard_location) = if options.is_inlined_outboard(outboard_size) {
216            if outboard_size == 0 {
217                (MemOrFile::empty(), OutboardLocation::NotNeeded)
218            } else {
219                let outboard = read_to_end(&self.outboard, 0, outboard_size as usize)?;
220                trace!("read outboard from file: {:?}", outboard.len());
221                (
222                    MemOrFile::Mem(outboard.clone()),
223                    OutboardLocation::Inline(outboard),
224                )
225            }
226        } else {
227            (MemOrFile::File(self.outboard), OutboardLocation::Owned)
228        };
229        // todo: notify the store that the state has changed to complete
230        Ok((
231            CompleteStorage { data, outboard },
232            EntryState::Complete {
233                data_location,
234                outboard_location,
235            },
236        ))
237    }
238
239    pub(super) fn current_size(&self) -> io::Result<u64> {
240        read_size(&self.sizes)
241    }
242
243    fn write_batch(&mut self, size: u64, batch: &[BaoContentItem]) -> io::Result<()> {
244        let tree = BaoTree::new(size, IROH_BLOCK_SIZE);
245        for item in batch {
246            match item {
247                BaoContentItem::Parent(parent) => {
248                    if let Some(offset) = tree.pre_order_offset(parent.node) {
249                        let o0 = offset * 64;
250                        self.outboard
251                            .write_all_at(o0, parent.pair.0.as_bytes().as_slice())?;
252                        self.outboard
253                            .write_all_at(o0 + 32, parent.pair.1.as_bytes().as_slice())?;
254                    }
255                }
256                BaoContentItem::Leaf(leaf) => {
257                    let o0 = leaf.offset;
258                    // divide by chunk size, multiply by 8
259                    let index = (leaf.offset >> (tree.block_size().chunk_log() + 10)) << 3;
260                    trace!(
261                        "write_batch f={:?} o={} l={}",
262                        self.data,
263                        o0,
264                        leaf.data.len()
265                    );
266                    self.data.write_all_at(o0, leaf.data.as_ref())?;
267                    let size = tree.size();
268                    self.sizes.write_all_at(index, &size.to_le_bytes())?;
269                }
270            }
271        }
272        Ok(())
273    }
274}

276fn read_size(size_file: &File) -> io::Result<u64> {
277    let len = size_file.metadata()?.len();
278    if len < 8 {
279        Ok(0)
280    } else {
281        let len = len & !7;
282        let mut buf = [0u8; 8];
283        size_file.read_exact_at(len - 8, &mut buf)?;
284        Ok(u64::from_le_bytes(buf))
285    }

288/// The storage for a bao file. This can be either in memory or on disk.
289///
290/// The two initial states `Initial` and `Loading` are used to coordinate the
291/// loading of the entry from the metadata database. Once that is complete,
292/// you should never see these states again.
293///
294/// From the remaining states you can get into `Poisoned` if there is an
295/// IO error during an operation.
296///
297/// `Poisioned` is also used once the handle is persisted and no longer usable.
298#[derive(derive_more::From, Default)]
299pub(crate) enum BaoFileStorage {
300    /// Initial state, we don't know anything yet.
301    #[default]
302    Initial,
303    /// Currently loading the entry from the metadata.
304    Loading,
305    /// There is no info about this hash in the metadata db.
306    NonExisting,
307    /// The entry is incomplete and in memory.
308    ///
309    /// Since it is incomplete, it must be writeable.
310    ///
311    /// This is used mostly for tiny entries, <= 16 KiB. But in principle it
312    /// can be used for larger sizes.
313    ///
314    /// Incomplete mem entries are *not* persisted at all. So if the store
315    /// crashes they will be gone.
316    PartialMem(PartialMemStorage),
317    /// The entry is incomplete and on disk.
318    Partial(PartialFileStorage),
319    /// The entry is complete. Outboard and data can come from different sources
320    /// (memory or file).
321    ///
322    /// Writing to this is a no-op, since it is already complete.
323    Complete(CompleteStorage),
324    /// We will get into that state if there is an io error in the middle of an operation,
325    /// or after the handle is persisted and no longer usable.
326    Poisoned,
327}

329impl fmt::Debug for BaoFileStorage {
330    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
331        match self {
332            BaoFileStorage::PartialMem(x) => x.fmt(f),
333            BaoFileStorage::Partial(x) => x.fmt(f),
334            BaoFileStorage::Complete(x) => x.fmt(f),
335            BaoFileStorage::Poisoned => f.debug_struct("Poisoned").finish(),
336            BaoFileStorage::Initial => f.debug_struct("Initial").finish(),
337            BaoFileStorage::Loading => f.debug_struct("Loading").finish(),
338            BaoFileStorage::NonExisting => f.debug_struct("NonExisting").finish(),
339        }
340    }
341}

343impl PartialMemStorage {
344    /// Converts this storage into a complete storage, using the given hash for
345    /// path names and the given options for decisions about inlining.
346    fn into_complete(self, ctx: &HashContext) -> io::Result<(CompleteStorage, EntryState<Bytes>)> {
347        let options = &ctx.global.options;
348        let hash = &ctx.id;
349        let size = self.current_size();
350        let outboard_size = raw_outboard_size(size);
351        let (data, data_location) = if options.is_inlined_data(size) {
352            let data: Bytes = self.data.to_vec().into();
353            (MemOrFile::Mem(data.clone()), DataLocation::Inline(data))
354        } else {
355            let data_path = options.path.data_path(hash);
356            let mut data_file = create_read_write(&data_path)?;
357            self.data.persist(&mut data_file)?;
358            (
359                MemOrFile::File(FixedSize::new(data_file, size)),
360                DataLocation::Owned(size),
361            )
362        };
363        let (outboard, outboard_location) = if ctx.global.options.is_inlined_outboard(outboard_size)
364        {
365            if outboard_size > 0 {
366                let outboard: Bytes = self.outboard.to_vec().into();
367                (
368                    MemOrFile::Mem(outboard.clone()),
369                    OutboardLocation::Inline(outboard),
370                )
371            } else {
372                (MemOrFile::empty(), OutboardLocation::NotNeeded)
373            }
374        } else {
375            let outboard_path = ctx.global.options.path.outboard_path(hash);
376            let mut outboard_file = create_read_write(&outboard_path)?;
377            self.outboard.persist(&mut outboard_file)?;
378            let outboard_location = if outboard_size == 0 {
379                OutboardLocation::NotNeeded
380            } else {
381                OutboardLocation::Owned
382            };
383            (MemOrFile::File(outboard_file), outboard_location)
384        };
385        Ok((
386            CompleteStorage { data, outboard },
387            EntryState::Complete {
388                data_location,
389                outboard_location,
390            },
391        ))
392    }
393}

395impl BaoFileStorage {
396    pub fn bitfield(&self) -> Bitfield {
397        match self {
398            BaoFileStorage::Initial => {
399                panic!("initial storage should not be used")
400            }
401            BaoFileStorage::Loading => {
402                panic!("loading storage should not be used")
403            }
404            BaoFileStorage::NonExisting => Bitfield::empty(),
405            BaoFileStorage::PartialMem(x) => x.bitfield.clone(),
406            BaoFileStorage::Partial(x) => x.bitfield.clone(),
407            BaoFileStorage::Complete(x) => Bitfield::complete(x.data.size()),
408            BaoFileStorage::Poisoned => {
409                panic!("poisoned storage should not be used")
410            }
411        }
412    }
413
414    pub(super) fn write_batch(
415        self,
416        batch: &[BaoContentItem],
417        bitfield: &Bitfield,
418        ctx: &HashContext,
419    ) -> io::Result<(Self, Option<EntryState<bytes::Bytes>>)> {
420        Ok(match self {
421            BaoFileStorage::NonExisting => {
422                Self::new_partial_mem().write_batch(batch, bitfield, ctx)?
423            }
424            BaoFileStorage::PartialMem(mut ms) => {
425                // check if we need to switch to file mode, otherwise write to memory
426                if max_offset(batch) <= ctx.global.options.inline.max_data_inlined {
427                    ms.write_batch(bitfield.size(), batch)?;
428                    let changes = ms.bitfield.update(bitfield);
429                    let new = changes.new_state();
430                    if new.complete {
431                        let (cs, update) = ms.into_complete(ctx)?;
432                        (cs.into(), Some(update))
433                    } else {
434                        let fs = ms.persist(ctx)?;
435                        let update = EntryState::Partial {
436                            size: new.validated_size,
437                        };
438                        (fs.into(), Some(update))
439                    }
440                } else {
441                    // *first* switch to file mode, *then* write the batch.
442                    //
443                    // otherwise we might allocate a lot of memory if we get
444                    // a write at the end of a very large file.
445                    //
446                    // opt: we should check if we become complete to avoid going from mem to partial to complete
447                    let mut fs = ms.persist(ctx)?;
448                    fs.write_batch(bitfield.size(), batch)?;
449                    let changes = fs.bitfield.update(bitfield);
450                    let new = changes.new_state();
451                    if new.complete {
452                        let size = new.validated_size.unwrap();
453                        let (cs, update) = fs.into_complete(size, &ctx.global.options)?;
454                        (cs.into(), Some(update))
455                    } else {
456                        let update = EntryState::Partial {
457                            size: new.validated_size,
458                        };
459                        (fs.into(), Some(update))
460                    }
461                }
462            }
463            BaoFileStorage::Partial(mut fs) => {
464                fs.write_batch(bitfield.size(), batch)?;
465                let changes = fs.bitfield.update(bitfield);
466                let new = changes.new_state();
467                if new.complete {
468                    let size = new.validated_size.unwrap();
469                    let (cs, update) = fs.into_complete(size, &ctx.global.options)?;
470                    (cs.into(), Some(update))
471                } else if changes.was_validated() {
472                    // we are still partial, but now we know the size
473                    let update = EntryState::Partial {
474                        size: new.validated_size,
475                    };
476                    (fs.into(), Some(update))
477                } else {
478                    (fs.into(), None)
479                }
480            }
481            BaoFileStorage::Complete(_) => {
482                // we are complete, so just ignore the write
483                // unless there is a bug, this would just write the exact same data
484                (self, None)
485            }
486            _ => {
487                // we are poisoned, so just ignore the write
488                (self, None)
489            }
490        })
491    }
492
493    /// Create a new mutable mem storage.
494    pub fn new_partial_mem() -> Self {
495        Self::PartialMem(Default::default())
496    }
497
498    /// Call sync_all on all the files.
499    #[allow(dead_code)]
500    fn sync_all(&self) -> io::Result<()> {
501        match self {
502            Self::Complete(_) => Ok(()),
503            Self::PartialMem(_) => Ok(()),
504            Self::NonExisting => Ok(()),
505            Self::Partial(file) => {
506                file.data.sync_all()?;
507                file.outboard.sync_all()?;
508                file.sizes.sync_all()?;
509                Ok(())
510            }
511            Self::Poisoned | Self::Initial | Self::Loading => {
512                // we are poisoned, so just ignore the sync
513                Ok(())
514            }
515        }
516    }
517
518    pub fn take(&mut self) -> Self {
519        std::mem::replace(self, BaoFileStorage::Poisoned)
520    }
521}

575impl BaoFileStorage {
576    pub async fn open(state: Option<EntryState<Bytes>>, ctx: &HashContext) -> io::Result<Self> {
577        let hash = &ctx.id;
578        let options = &ctx.global.options;
579        Ok(match state {
580            Some(EntryState::Complete {
581                data_location,
582                outboard_location,
583            }) => {
584                let data = match data_location {
585                    DataLocation::Inline(data) => MemOrFile::Mem(data),
586                    DataLocation::Owned(size) => {
587                        let path = options.path.data_path(hash);
588                        let file = std::fs::File::open(&path)?;
589                        MemOrFile::File(FixedSize::new(file, size))
590                    }
591                    DataLocation::External(paths, size) => {
592                        let Some(path) = paths.into_iter().next() else {
593                            return Err(io::Error::other("no external data path"));
594                        };
595                        let file = std::fs::File::open(&path)?;
596                        MemOrFile::File(FixedSize::new(file, size))
597                    }
598                };
599                let outboard = match outboard_location {
600                    OutboardLocation::NotNeeded => MemOrFile::empty(),
601                    OutboardLocation::Inline(data) => MemOrFile::Mem(data),
602                    OutboardLocation::Owned => {
603                        let path = options.path.outboard_path(hash);
604                        let file = std::fs::File::open(&path)?;
605                        MemOrFile::File(file)
606                    }
607                };
608                Self::new_complete(data, outboard)
609            }
610            Some(EntryState::Partial { .. }) => Self::new_partial_file(ctx).await?,
611            None => Self::NonExisting,
612        })
613    }
614
615    /// Create a new bao file handle with a partial file.
616    pub(super) async fn new_partial_file(ctx: &HashContext) -> io::Result<Self> {
617        let hash = &ctx.id;
618        let options = ctx.global.options.clone();
619        let storage = PartialFileStorage::load(hash, &options.path)?;
620        Ok(if storage.bitfield.is_complete() {
621            let size = storage.bitfield.size;
622            let (storage, entry_state) = storage.into_complete(size, &options)?;
623            debug!("File was reconstructed as complete");
624            ctx.global.db.set(*hash, entry_state).await?;
625            storage.into()
626        } else {
627            storage.into()
628        })
629    }

631    /// Create a new complete bao file handle.
632    pub fn new_complete(
633        data: MemOrFile<Bytes, FixedSize<File>>,
634        outboard: MemOrFile<Bytes, File>,
635    ) -> Self {
636        CompleteStorage { data, outboard }.into()
637    }

640impl BaoFileHandle {
641    /// Complete the handle
642    pub fn complete(
643        &self,
644        data: MemOrFile<Bytes, FixedSize<File>>,
645        outboard: MemOrFile<Bytes, File>,
646    ) {
647        self.send_if_modified(|guard| {
648            let needs_complete = match guard {
649                BaoFileStorage::NonExisting => true,
650                BaoFileStorage::Complete(_) => false,
651                BaoFileStorage::PartialMem(_) => true,
652                BaoFileStorage::Partial(_) => true,
653                _ => false,
654            };
655            if needs_complete {
656                *guard = BaoFileStorage::Complete(CompleteStorage { data, outboard });
657                true
658            } else {
659                true
660            }
661        });
662    }

665impl PartialMemStorage {
666    /// Persist the batch to disk.
667    fn persist(self, ctx: &HashContext) -> io::Result<PartialFileStorage> {
668        let options = &ctx.global.options.path;
669        let hash = &ctx.id;
670        ctx.global.protect.protect(
671            *hash,
672            [
673                BaoFilePart::Data,
674                BaoFilePart::Outboard,
675                BaoFilePart::Sizes,
676                BaoFilePart::Bitfield,
677            ],
678        );
679        let mut data = create_read_write(options.data_path(hash))?;
680        let mut outboard = create_read_write(options.outboard_path(hash))?;
681        let mut sizes = create_read_write(options.sizes_path(hash))?;
682        self.data.persist(&mut data)?;
683        self.outboard.persist(&mut outboard)?;
684        self.size.persist(&mut sizes)?;
685        data.sync_all()?;
686        outboard.sync_all()?;
687        sizes.sync_all()?;
688        Ok(PartialFileStorage {
689            data,
690            outboard,
691            sizes,
692            bitfield: self.bitfield,
693        })
694    }
695}

701impl BaoFileStorageSubscriber {
702    pub fn new(receiver: watch::Receiver<BaoFileStorage>) -> Self {
703        Self { receiver }
704    }
705
706    /// Forward observed *values* to the given sender
707    ///
708    /// Returns an error if sending fails, or if the last sender is dropped
709    pub async fn forward(mut self, mut tx: mpsc::Sender<Bitfield>) -> anyhow::Result<()> {
710        let value = self.receiver.borrow().bitfield();
711        tx.send(value).await?;
712        loop {
713            self.update_or_closed(&mut tx).await?;
714            let value = self.receiver.borrow().bitfield();
715            tx.send(value.clone()).await?;
716        }
717    }
718
719    /// Forward observed *deltas* to the given sender
720    ///
721    /// Returns an error if sending fails, or if the last sender is dropped
722    #[allow(dead_code)]
723    pub async fn forward_delta(mut self, mut tx: mpsc::Sender<Bitfield>) -> anyhow::Result<()> {
724        let value = self.receiver.borrow().bitfield();
725        let mut old = value.clone();
726        tx.send(value).await?;
727        loop {
728            self.update_or_closed(&mut tx).await?;
729            let new = self.receiver.borrow().bitfield();
730            let diff = old.diff(&new);
731            if diff.is_empty() {
732                continue;
733            }
734            tx.send(diff).await?;
735            old = new;
736        }
737    }
738
739    async fn update_or_closed(&mut self, tx: &mut mpsc::Sender<Bitfield>) -> anyhow::Result<()> {
740        tokio::select! {
741            _ = tx.closed() => {
742                // the sender is closed, we are done
743                Err(irpc::channel::SendError::ReceiverClosed.into())
744            }
745            e = self.receiver.changed() => Ok(e?),
746        }
747    }
